// 4. WIDGET UPDATE FREQUENCY
// ============================================

/* File: android/app/src/main/res/xml/clock_widget_info.xml

Update every 30 seconds (more frequent):
<appwidget-provider
    ...
    android:updatePeriodMillis="30000"
    ...
/>

Update every 5 minutes (less battery drain):
<appwidget-provider
    ...
    android:updatePeriodMillis="300000"
    ...
/>

Manual update only (no auto-refresh):
<appwidget-provider
    ...
    android:updatePeriodMillis="0"
    ...
/>

Note: The system respects a minimum of 30 minutes for battery efficiency.
TextClock overrides this with real-time updates anyway.
*/

// 6. ADVANCED FLUTTER: ANIMATION EXAMPLE
// ============================================

/*
Add animated rotation to the clock display:

In lib/main.dart _ClockHomeState, add:
late AnimationController _rotationController;

@override
void initState() {
  super.initState();
  _timeStream = Stream.periodic(
    const Duration(seconds: 1),
    (_) => DateTime.now(),
  );
  
  _rotationController = AnimationController(
    duration: const Duration(seconds: 60),
    vsync: this,
  )..repeat();
}

@override
void dispose() {
  _rotationController.dispose();
  super.dispose();
}

Then wrap the time display with RotationTransition:
RotationTransition(
  turns: _rotationController,
  child: Text(
    '$hour:$minute',
    style: TextStyle(
      fontSize: 72,
      fontWeight: FontWeight.bold,
      color: colorScheme.primary,
    ),
  ),
)
*/

// 7. HOME WIDGET COMMUNICATION
// ============================================

/*
To send data from Flutter app to widget:

1. Add to pubspec.yaml:
   home_widget: ^0.4.3

2. In lib/main.dart, add:
   import 'package:home_widget/home_widget.dart';

   In main():
   await HomeWidget.setAppGroupId(
     appGroupId: 'com.example.clock_widget',
   );

3. To send data to widget:
   await HomeWidget.saveWidgetData<String>(
     'key',
     'value',
   );

   // Trigger widget update
   await HomeWidget.updateWidget(
     name: 'ClockWidgetProvider',
     iOSName: 'ClockWidget',
   );

4. In ClockWidgetProvider.kt, read data:
   val sharedPref = context.getSharedPreferences(
     "FlutterSharedPreferences",
     Context.MODE_PRIVATE,
   )
   val value = sharedPref.getString("flutter.key", "")
*/

// 9. RESPONSIVE TEXT SIZING
// ============================================

/*
Make text sizes responsive to widget size:

In widget_layout.xml, use sp (scale-independent pixels):
- Very Small Widget: textSize="32sp" (instead of 56sp)
- Normal Widget: textSize="56sp"
- Large Widget: textSize="72sp"

Or use multiple layout files:
- layout/widget_layout.xml (default - 4x3)
- layout-land/widget_layout.xml (landscape - 6x3)
- layout-small/widget_layout.xml (small - 2x2)
*/
